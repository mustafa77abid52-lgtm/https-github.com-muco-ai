# muco_ai_core.py
import numpy as np
import pandas as pd

from transformers import pipeline
import cv2
import matplotlib.pyplot as plt

class MUCO_AI:
    def __init__(self):
        self.models = {}
        self.nlp_processor = pipeline("text-generation", model="gpt2")
        
    def financial_analysis(self, data):
        """Finansal veri analizi algoritması"""
        # Zaman serisi analizi
        df = pd.DataFrame(data)
        
        # Teknik göstergeler
        df['SMA_20'] = df['close'].rolling(window=20).mean()
        df['SMA_50'] = df['close'].rolling(window=50).mean()
        df['RSI'] = self.calculate_rsi(df['close'])
        
        return df
    
    def calculate_rsi(self, prices, period=14):
        """RSI hesaplama"""
        delta = prices.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    def content_generator(self, topic, content_type="article"):
        """İçerik üretici AI"""
        prompts = {
            "article": f"{topic} hakkında detaylı bir makale:",
            "video_script": f"{topic} konulu video senaryosu:",
            "social_media": f"{topic} için sosyal medya içeriği:"
        }
        
        generated = self.nlp_processor(
            prompts[content_type],
            max_length=500,
            temperature=0.7
        )
        
        return generated[0]['generated_text']
    
    def design_suggestions(self, brand_info):
        """Logo ve tasarım önerileri"""
        color_palettes = {
            "technology": ["#2563eb", "#3b82f6", "#60a5fa", "#93c5fd"],
            "finance": ["#059669", "#10b981", "#34d399", "#6ee7b7"],
            "creative": ["#7c3aed", "#8b5cf6", "#a78bfa", "#c4b5fd"]
        }
        
        suggestions = {
            "colors": color_palettes.get(brand_info.get("sector", "creative")),
            "fonts": ["Inter", "Roboto", "Montserrat", "Open Sans"],
            "logo_types": ["Minimal", "Mascot", "Wordmark", "Abstract"]
        }
        
        return suggestions

import React, { useState } from 'react';
import axios from 'axios';

  const [topic, setTopic] = useState('');
  const [generatedContent, setGeneratedContent] = useState('');
  const [designData, setDesignData] = useState(null);

  const generateContent = async () => {
    try {
      const response = await axios.post('http://localhost:8000/generate-content', {
        topic: topic,
        content_type: 'article'
      });
      setGeneratedContent(response.data.content);
    } catch (error) {
      console.error('Error:', error);
    }
  };

  const getDesignSuggestions = async () => {
    try {
      const response = await axios.post('http://localhost:8000/design-suggestions', {
        company_name: 'MUCO',
        sector: 'technology',
        preferences: { style: 'modern' }
      });
      setDesignData(response.data.suggestions);
    } catch (error) {
      console.error('Error:', error);
    }
  };

  return (
    <div className="muco-dashboard">
      <header className="dashboard-header">

        """Kazanç hesaplama algoritması"""
        # Örnek formül: (İzlenme x CPM) + (Tıklama x CPC) + Abonelik

            metrics.get("clicks", 0) * metrics.get("cpc", 0.1) +
            metrics.get("subscriptions", 0) * 9.99
