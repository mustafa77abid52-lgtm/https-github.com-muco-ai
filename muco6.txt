    def check_system_requirements(self):

        # Minimum RAM kontrolÃ¼ (2GB)
        if system == "Windows":
            import ctypes
            kernel32 = ctypes.windll.kernel32
            ctypes.windll.kernel32.GetPhysicallyInstalledSystemMemory()
            memory = kernel32.GetPhysicallyInstalledSystemMemory()
            memory_gb = memory / (1024**3)
        else:
            with open('/proc/meminfo') as f:
                for line in f:
                    if 'MemTotal' in line:
                        memory = int(line.split()[1])
                        memory_gb = memory / (1024**2)
                        break
        
        if memory_gb < 2:
            raise Exception(f"Yetersiz RAM: {memory_gb:.1f}GB (Minimum 2GB gerekli)")
        
        self.log_message(f"RAM: {memory_gb:.1f}GB âœ“")
        
        # Python versiyonu kontrolÃ¼
        python_version = sys.version_info
        if python_version < (3, 8):
            raise Exception(f"Python versiyonu Ã§ok eski: {python_version.major}.{python_version.minor} (Minimum 3.8 gerekli)")
        
        self.log_message(f"Python: {python_version.major}.{python_version.minor}.{python_version.micro} âœ“")
        
        # Disk alanÄ± kontrolÃ¼ (1GB)
        if system == "Windows":
            import ctypes
            free_bytes = ctypes.c_ulonglong(0)
            ctypes.windll.kernel32.GetDiskFreeSpaceExW(
                ctypes.c_wchar_p(str(self.setup_dir)), 
                None, None, ctypes.pointer(free_bytes)
            )
            free_gb = free_bytes.value / (1024**3)
        else:
            import shutil
            free_gb = shutil.disk_usage(self.setup_dir).free / (1024**3)
        
        if free_gb < 1:
            raise Exception(f"Yetersiz disk alanÄ±: {free_gb:.1f}GB (Minimum 1GB gerekli)")

        """Ä°nternetten AI model verilerini topla"""

        return features.astype(np.float32)

class MobileAIArchitecture(nn.Module):
    """Mobil iÃ§in optimize edilmiÅŸ AI mimarisi"""
    
    def __init__(self, input_size=10, hidden_size=64, output_size=5):
        super().__init__()
        
        # Hafif mimari
        self.encoder = nn.Sequential(
            nn.Linear(input_size, hidden_size),
            nn.ReLU(),
            nn.Dropout(0.1),
            nn.Linear(hidden_size, hidden_size // 2),
            nn.ReLU(),
            nn.Dropout(0.1),
            nn.Linear(hidden_size // 2, output_size)
        )
        
        # Quantization hazÄ±rlÄ±ÄŸÄ±
        self.quant = torch.quantization.QuantStub()
        self.dequant = torch.quantization.DeQuantStub()
    
    def forward(self, x):
        x = self.quant(x)
        x = self.encoder(x)
        x = self.dequant(x)
        return x
    
    def convert_to_tflite(self, output_path="mobile_model.tflite"):
        """Modeli TensorFlow Lite formatÄ±na dÃ¶nÃ¼ÅŸtÃ¼r"""

    model = MobileAIArchitecture()

    criterion = nn.MSELoss()
    optimizer = optim.Adam(model.parameters(), lr=0.001)

    # EÄŸitim dÃ¶ngÃ¼sÃ¼
    num_epochs = 10
    for epoch in range(num_epochs):
        total_loss = 0
        model.train()
        
        progress_bar = tqdm(dataloader, desc=f"Epoch {epoch+1}/{num_epochs}")
        for batch in progress_bar:
            features = batch['features']
            
            # Forward pass
            outputs = model(features)
            
            # Basit loss hesaplama
            loss = criterion(outputs, features[:, :5])
            
            # Backward pass
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            
            total_loss += loss.item()
            progress_bar.set_postfix({'loss': loss.item()})
